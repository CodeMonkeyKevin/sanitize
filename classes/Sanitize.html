 77:     fragment = Hpricot(html)

 79:     fragment.traverse_element do |node|
 80:       if node.bogusetag? || node.doctype? || node.procins? || node.xmldecl?
 81:         node.swap('')
 82:         next
 83:       end
 84: 
 85:       if node.comment?
 86:         node.swap('') unless @config[:allow_comments]
 87:       elsif node.elem?
 88:         name = node.name.downcase
 89: 
 90:         # Delete any element that isn't in the whitelist.
 91:         unless @config[:elements].include?(name)
 92:           node.parent.replace_child(node, node.children)
 93:           next
 94:         end
 95: 
 96:         if @config[:attributes].has_key?(name)
 97:           # Delete any attribute that isn't in the whitelist for this element.
 98:           node.raw_attributes.delete_if do |key, value|
 99:             !@config[:attributes][name].include?(key.downcase)
100:           end
101: 
102:           # Delete remaining attributes that use unacceptable protocols.
103:           if @config[:protocols].has_key?(name)
104:             protocol = @config[:protocols][name]
105: 
106:             node.raw_attributes.delete_if do |key, value|
107:               next false unless protocol.has_key?(key)
108: 
109:               if value.downcase =~ /^([^:]+)(?:\:|&#0*58;|&#x0*3a;)/
110:                 !protocol[key].include?($1.downcase)
111:               else
112:                 !protocol[key].include?(:relative)
113:               end
114:             end
115:           end
116:         else
117:           # Delete all attributes from elements with no whitelisted
118:           # attributes.
119:           node.raw_attributes = {}
120:         end
121: 
122:         # Add required attributes.
123:         if @config[:add_attributes].has_key?(name)
124:           node.raw_attributes.merge!(@config[:add_attributes][name])
125:         end
126:       end
127:     end
